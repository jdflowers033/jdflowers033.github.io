  <script>

    // -------------------------------------------------------

    // Target moment:

    // May 8, 2026, 8:00 AM Central Time (CDT, UTC-5)

    // That corresponds to 2026-05-08T13:00:00Z

    // -------------------------------------------------------

    const targetDate = new Date("2026-05-08T13:00:00Z");


    const daysEl = document.getElementById("days");

    const hoursEl = document.getElementById("hours");

    const minutesEl = document.getElementById("minutes");

    const secondsEl = document.getElementById("seconds");

    const messageEl = document.getElementById("message");


    function pad(num) {

      return num.toString().padStart(2, "0");

    }


    function applyFlip(el) {

      el.classList.add("flip");

      setTimeout(() => el.classList.remove("flip"), 200);

    }


    function updateCountdown() {

      const now = new Date();

      const diffMs = targetDate - now;


      if (diffMs <= 0) {

        // Event has started/passed

        daysEl.textContent = "0";

        hoursEl.textContent = "00";

        minutesEl.textContent = "00";

        secondsEl.textContent = "00";

        messageEl.textContent = "The event time has arrived!";

        return;

      }


      const totalSeconds = Math.floor(diffMs / 1000);


      const days = Math.floor(totalSeconds / (24 * 60 * 60));

      const hours = Math.floor(

        (totalSeconds % (24 * 60 * 60)) / (60 * 60)

      );

      const minutes = Math.floor((totalSeconds % (60 * 60)) / 60);

      const seconds = totalSeconds % 60;


      // Only animate if the value actually changes

      if (daysEl.textContent !== String(days)) {

        daysEl.textContent = days;

        applyFlip(daysEl);

      }

      const hoursStr = pad(hours);

      if (hoursEl.textContent !== hoursStr) {

        hoursEl.textContent = hoursStr;

        applyFlip(hoursEl);

      }

      const minutesStr = pad(minutes);

      if (minutesEl.textContent !== minutesStr) {

        minutesEl.textContent = minutesStr;

        applyFlip(minutesEl);

      }

      const secondsStr = pad(seconds);

      if (secondsEl.textContent !== secondsStr) {

        secondsEl.textContent = secondsStr;

        applyFlip(secondsEl);

      }


      messageEl.textContent = ""; // clear any previous message

    }


    // Initial draw

    updateCountdown();

    // Update every second

    setInterval(updateCountdown, 1000);

  </script>
